// government-tracker.js - Complete Single File // Run: node government-tracker.js // Visit: http://localhost:3000 const express = require('express'); const { v4: uuidv4 } = require('uuid'); const multer = require('multer'); const path = require('path'); const app = express(); app.use(express.json()); app.use(express.static('public')); // File upload setup for photos const storage = multer.memoryStorage(); const upload = multer({ storage, limits: { fileSize: 5 * 1024 * 1024 }, // 5MB fileFilter: (req, file, cb) => { const allowedTypes = ['image/jpeg', 'image/png', 'image/gif']; if (allowedTypes.includes(file.mimetype)) { cb(null, true); } else { cb(new Error('Only images allowed')); } } }); // In-memory storage const cities = [ { id: '1', name: 'Springfield', state: 'IL', population: 150000, avgResponseDays: { pothole: 12, streetlight: 8, noise: 5, garbage: 4, other: 10 } }, { id: '2', name: 'Shelbyville', state: 'IL', population: 80000, avgResponseDays: { pothole: 18, streetlight: 10, noise: 7, garbage: 6, other: 14 } }, { id: '3', name: 'Capital City', state: 'IL', population: 250000, avgResponseDays: { pothole: 9, streetlight: 6, noise: 4, garbage: 3, other: 8 } }, { id: '4', name: 'Oakland', state: 'CA', population: 440000, avgResponseDays: { pothole: 45, streetlight: 30, noise: 20, garbage: 15, other: 35 } }, { id: '5', name: 'Austin', state: 'TX', population: 960000, avgResponseDays: { pothole: 14, streetlight: 9, noise: 6, garbage: 5, other: 12 } } ]; const requests = []; const categories = [ { id: 'pothole', name: '🕳️ Pothole', icon: '🕳️', avgTime: 14 }, { id: 'streetlight', name: '💡 Broken Streetlight', icon: '💡', avgTime: 8 }, { id: 'noise', name: '🔊 Noise Complaint', icon: '🔊', avgTime: 5 }, { id: 'garbage', name: '🗑️ Missed Garbage', icon: '🗑️', avgTime: 4 }, { id: 'graffiti', name: '🎨 Graffiti', icon: '🎨', avgTime: 10 }, { id: 'tree', name: '🌳 Fallen Tree', icon: '🌳', avgTime: 12 }, { id: 'water', name: '💧 Water Leak', icon: '💧', avgTime: 8 }, { id: 'other', name: '📋 Other', icon: '📋', avgTime: 10 } ]; // Helper functions function getCityById(id) { return cities.find(c => c.id === id); } function getCategoryById(id) { return categories.find(c => c.id === id); } function calculateResponseTime(createdAt, resolvedAt) { if (!resolvedAt) return null; const created = new Date(createdAt); const resolved = new Date(resolvedAt); const diffDays = Math.ceil((resolved - created) / (1000 * 60 * 60 * 24)); return diffDays; } function getStatusColor(status) { const colors = { submitted: '#f59e0b', investigating: '#3b82f6', in_progress: '#8b5cf6', resolved: '#10b981', closed: '#6b7280' }; return colors[status] || '#6b7280'; } // Routes // Get all cities app.get('/api/cities', (req, res) => { res.json(cities); }); // Get city rankings app.get('/api/cities/rankings', (req, res) => { const rankings = cities.map(city => { const cityRequests = requests.filter(r => r.cityId === city.id); const avgResponseByCategory = {}; categories.forEach(cat => { const catRequests = cityRequests.filter(r => r.category === cat.id && r.resolvedAt); if (catRequests.length > 0) { const avg = catRequests.reduce((sum, r) => sum + r.responseTime, 0) / catRequests.length; avgResponseByCategory[cat.id] = Math.round(avg); } else { avgResponseByCategory[cat.id] = city.avgResponseDays[cat.id] || 10; } }); const totalRequests = cityRequests.length; const resolvedRequests = cityRequests.filter(r => r.resolvedAt).length; const resolutionRate = totalRequests > 0 ? Math.round((resolvedRequests / totalRequests) * 100) : 0; return { ...city, avgResponseByCategory, totalRequests, resolutionRate, pendingRequests: totalRequests - resolvedRequests }; }); rankings.sort((a, b) => a.avgResponseDays.pothole - b.avgResponseDays.pothole); res.json(rankings); }); // Get city statistics app.get('/api/cities/:cityId/stats', (req, res) => { const { cityId } = req.params; const city = getCityById(cityId); if (!city) return res.status(404).json({ error: 'City not found' }); const cityRequests = requests.filter(r => r.cityId === cityId); const stats = { city, totalRequests: cityRequests.length, openRequests: cityRequests.filter(r => !r.resolvedAt).length, resolvedRequests: cityRequests.filter(r => r.resolvedAt).length, avgResponseTime: 0, byCategory: {}, recentRequests: cityRequests.slice(-10).reverse() }; const resolvedRequests = cityRequests.filter(r => r.resolvedAt); if (resolvedRequests.length > 0) { const totalDays = resolvedRequests.reduce((sum, r) => sum + r.responseTime, 0); stats.avgResponseTime = Math.round(totalDays / resolvedRequests.length); } categories.forEach(cat => { const catRequests = cityRequests.filter(r => r.category === cat.id); const catResolved = catRequests.filter(r => r.resolvedAt); stats.byCategory[cat.id] = { total: catRequests.length, open: catRequests.filter(r => !r.resolvedAt).length, resolved: catResolved.length, avgResponseTime: catResolved.length > 0 ? Math.round(catResolved.reduce((sum, r) => sum + r.responseTime, 0) / catResolved.length) : city.avgResponseDays[cat.id] || 10 }; }); res.json(stats); }); // Submit new request app.post('/api/requests', upload.single('photo'), (req, res) => { const { cityId, category, title, description, address, name, email } = req.body; const id = uuidv4(); const city = getCityById(cityId); if (!city) return res.status(400).json({ error: 'Invalid city' }); const request = { id, cityId, category, title: title || `${getCategoryById(category)?.name} at ${address || 'unknown location'}`, description, address, status: 'submitted', statusHistory: [{ status: 'submitted', timestamp: new Date().toISOString() }], submittedBy: { name, email }, createdAt: new Date().toISOString(), resolvedAt: null, responseTime: null, photo: req.file ? `data:${req.file.mimetype};base64,${req.file.buffer.toString('base64')}` : null, updates: [] }; requests.push(request); res.json({ id, trackingUrl: `http://localhost:3000/track/${id}` }); }); // Get all requests for a city app.get('/api/cities/:cityId/requests', (req, res) => { const { cityId } = req.params; const { status, category, page = 1, limit = 20 } = req.query; let filtered = requests.filter(r => r.cityId === cityId); if (status) { if (status === 'open') filtered = filtered.filter(r => !r.resolvedAt); if (status === 'resolved') filtered = filtered.filter(r => r.resolvedAt); } if (category && category !== 'all') { filtered = filtered.filter(r => r.category === category); } filtered.sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt)); const start = (parseInt(page) - 1) * parseInt(limit); const paginated = filtered.slice(start, start + parseInt(limit)); res.json({ requests: paginated, total: filtered.length, page: parseInt(page), totalPages: Math.ceil(filtered.length / parseInt(limit)) }); }); // Get single request app.get('/api/requests/:id', (req, res) => { const request = requests.find(r => r.id === req.params.id); if (!request) return res.status(404).json({ error: 'Request not found' }); const city = getCityById(request.cityId); const category = getCategoryById(request.category); // Calculate days pending const daysPending = request.resolvedAt ? null : Math.ceil((new Date() - new Date(request.createdAt)) / (1000 * 60 * 60 * 24)); // Compare to city average const cityAvg = city.avgResponseDays[request.category] || 10; const isBehindAverage = daysPending && daysPending > cityAvg; res.json({ ...request, cityName: city.name, categoryName: category.name, daysPending, cityAvgResponse: cityAvg, isBehindAverage, statusColor: getStatusColor(request.status) }); }); // Update request status (admin function) app.put('/api/requests/:id/status', (req, res) => { const { id } = req.params; const { status, note } = req.body; const request = requests.find(r => r.id === id); if (!request) return res.status(404).json({ error: 'Request not found' }); request.status = status; request.statusHistory.push({ status, note, timestamp: new Date().toISOString() }); if (status === 'resolved' && !request.resolvedAt) { request.resolvedAt = new Date().toISOString(); request.responseTime = calculateResponseTime(request.createdAt, request.resolvedAt); } res.json(request); }); // Add update to request app.post('/api/requests/:id/updates', (req, res) => { const { id } = req.params; const { message, isOfficial } = req.body; const request = requests.find(r => r.id === id); if (!request) return res.status(404).json({ error: 'Request not found' }); request.updates.push({ id: uuidv4(), message, isOfficial: isOfficial || false, timestamp: new Date().toISOString() }); res.json(request); }); // Public tracking page route app.get('/track/:id', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'track.html')); }); // Frontend main page app.get('/', (req, res) => { res.send(` Local Government Request Tracker - Hold Your City Accountable
